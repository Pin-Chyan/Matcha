mongo syntax

to access database and will create one if not exist
    > use 'databse_name'

to check database currentluy access or in use
    > db 

to delete current database accessed
    > db.database()

to add information to database
    > db.players.insert(<info>)
    players is a collection of data
    if a collection of the specified name is not present mongo will create one
    to insert data
    this is the data input format
    > db.players.insert(
        {
            "information" = "name of someone",
            "age" = 46,
        },
        {
            "information" = "multiple data input",
            "age" = 83,
        }
    )

to view all  information
    >show collections

to find information
    >db.player.find(
        {"name" : "kevin", "age" : 209} // seach using and
        /////////////////////////////////
        $or:[                           // search using or
            {"name" : "kevin"},
            {"name" : "bob"}
        ]
        //////////////////////////////
        {"age" : {$gt:30}}              // search for all where age is $gt(greater than in mongo) greater than  
    )
    add .pretty() for more organised view
        >db.player.find().pretty()
    to return specified information
    >db.player.find(
        {"name" : "kevin"},
        {"name":1, _id:0, "age":0} // to return specific info the var name will be refferenced to 1 or 0 , 1 returns , 0 does not 
    ).limit(4) /// add limit to limit the returned amm of data
 
to remove information
    >db.player.remove(
        {"player" : "kevin"}
    )
    removes the document in collection player where "player" equals "kevin"

to update information
    >db.players.update(
        {"name" : "kevin"},
        {
            "name" : "bob",
            "age" : 209,
        }
    )
    updates name and age of document where name equals kevin

to drop individual collecitons in database
    >db.players.drop()

to index for faster search
    >db.players.ensureIndex({"name":1}) // indexes players collection using "name"
